SELECT a.*, b.i_board, nvl(b.like_cnt, 0) FROM t_board a
LEFT JOIN (
    SELECT i_board, count(i_board) AS LIKE_CNT
    FROM t_board_like
    GROUP BY i_board
) b
ON a.i_board = b.i_board;
